		zone_research_engineering
		zone_research_society
		zone_research_unity
		zone_trade
		zone_fortress
		zone_urban_food
		zone_urban_minerals
		zone_urban_energy
		# zone_foundry_wilderness
		# zone_unity_wilderness
		# zone_research_wilderness
		# zone_research_unity_wilderness
		# zone_fortress_wilderness
		# zone_trade_wilderness
	}

	exclude = {
		zone_default
		zone_minerals
		zone_food
		zone_energy
		oxr_mdlc_origin_wm_zone_default
	}


	unlock = { # planet scope
		custom_tooltip = {
			fail_text = zone_city_02_prereq
			OR = {
				is_capital = yes
				AND = {
					is_capital = no
					exists = owner
					owner = {
						has_tech_or_functional_civic = {
							TECH = tech_colonial_centralization
						}
					}
					OR = {
						has_upgraded_capital = yes
						oxr_mdlc_has_upgraded_capital = yes
					}
					
				}
			}
		}
	}
}
